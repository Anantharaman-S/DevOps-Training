#My Project
## Git Commands:
   
# Configure Git
git config --global user.name "Your Name"
git config --global user.email "your.email@example.com"

# Create project folder and initialize repo
mkdir my-project
cd my-project
git init

# Create initial file and commit
echo "# My Project" > trial.txt
git add trial.txt
git commit -m "Initial commit"

# Create and switch to feature branch
git branch feature/update-trial
git checkout feature/update-trial

# Update trial.txt and commit
echo "## Description" >> trial.txt
git add trial.txt
git commit -m "Update trial.txt with description"

# Merge back to main
git checkout main
git merge feature/update-trial

# Connect to GitHub remote (replace <url>)
git remote add origin https://github.com/<your-username>/<repo>.git
git remote -v

# Push main branch
git push origin main

# Create another branch and commit
git checkout -b feature/new-note
echo "# Extra note" >> trial.txt
git add trial.txt
git commit -m "Add extra note"
git push origin feature/new-note

# Fetch updates
git fetch

# Cherry-pick a commit (replace <commit>)
git checkout main
git cherry-pick <commit>

# Stash workflow
echo "Temporary change" >> trial.txt
git stash
git checkout -b hotfix/temp
git stash pop
git add trial.txt
git commit -m "Apply stashed change"

# Revert a commit (replace <bad_commit>)
git revert <bad_commit>

# Reset to previous commit (replace <good_commit>)
git reset --hard <good_commit>

# Tag a release
git tag v1.0.0
git push origin v1.0.0
